Discretizing 7.33

\begin{align} & \partial_t u ^\mu(t, x) + H^*(t,x, \mu_t, u^{\mu}(t, \cdot)) = 0\\ & \partial_t u ^\mu(t, x) + \lambda_t(x, \cdot, \mu_t, \hat{\alpha}) u^{\mu}(t, \cdot) + f(x) = 0\\ & \frac{u ^\mu(t+ \delta t, x) - u ^\mu(t, x)}{t} + \frac{P_{x, \cdot}^{\mu_t, \hat{\alpha}}(\delta t) - P_{x, \cdot}^{\mu_t, \hat{\alpha}}(0)}{\delta t} u^{\mu}(t + \delta t, \cdot) + f(x) = 0 \\ & u ^\mu(t, x) = P_{x, \cdot}^{\mu_t, \hat{\alpha}}(\delta t)u^{\mu}(t + \delta t, \cdot) + f(x) \delta t \end{align}

Discretizing 7.36

\begin{align} & \partial_t \mu_t(x) = \mu_t(\cdot) \lambda_t(\cdot, x, \mu_t, \hat{\alpha})\\ & \frac{\mu_{t+ \delta t}(x) - \mu_t(x)}{\delta t} = \mu_t(\cdot) \frac{P_{\cdot, x}^{\mu_t, \hat{\alpha}}(\delta t) - P_{\cdot, x}^{\mu_t, \hat{\alpha}}(0)}{\delta t} \\ & \mu_{t + \delta t}(x) = \mu_t(\cdot) P_{\cdot, x}^{\mu_t, \hat{\alpha}}(\delta t) \end{align}

def get_qmat(, a,
             q_inf_U=0.3,
             q_inf_D=0.4,
             q_rec_D=0.5,
             q_rec_U=0.4,
             =0.8,
             _UU=0.3,
             _UD=0.4,
             _DU=0.3,
             _DD=0.4,
             _H=0.6):
    '''
    Get infinitesimal generator matrix given measure and action
    Input:
    : dictionary or pd.series like {'DI': 0.25, 'DS': 0.25, 'UI': 0.25, 'US': 0.25}
    a: action in [0, 1]
    
    Output:
    Q: 4 X 4 matrix
    '''
    if a == 0:
        Q = np.array([
            [-q_rec_D, q_rec_D, 0, 0],
            [_H*q_inf_D + _DD*['DI'] + _UD*['UI'], - (_H*q_inf_D + _DD*['DI'] + _UD*['UI']), 0, 0],
            [0, 0, -q_rec_U , q_rec_U],
            [0, 0, _H*q_inf_U + _UU*['UI']+ _DU*['DI'], -(_H*q_inf_U + _UU*['UI']+ _DU*['DI'])]   
            ])
    elif a == 1:
        Q = np.array([
            [-(q_rec_D+), q_rec_D, , 0],
            [_H*q_inf_D + _DD*['DI'] + _UD*['UI'], - (_H*q_inf_D + _DD*['DI'] + _UD*['UI'] + ), 0, ],
            [, 0, -(q_rec_U+) , q_rec_U],
            [0, , _H*q_inf_U + _UU*['UI']+ _DU*['DI'], -(_H*q_inf_U + _UU*['UI']+ _DU*['DI']+)]   
            ])
    return Q

def get_f(k_D= 0.3, k_I= 0.5):
    '''
    Running cost function
    '''
    k = np.array([k_D + k_I, k_D, k_I, 0])
    return k

def solve_hjb(_flow, param_q, param_f):
    '''
    Solve discrete dynamic programming given a fixed 
    '''
    a_space = [0, 1]
    u = pd.DataFrame(np.zeros(_flow.shape),
                     index=_flow.index, columns=_flow.columns)             # initialize value function
    a_flow = pd.DataFrame(np.zeros(_flow.shape),
                          index=_flow.index, columns=_flow.columns)        # initialize action flow
    dt = _flow.index[1]                                                     # get time step
    
    f = get_f(**param_f)
    for t in reversed(range(len(u)-1)):
         = _flow.iloc[t]
        Q = [get_qmat(, a, **param_q) for a in a_space]                     # Q matrix by  and action
        P = [expm(q*dt) for q in Q]
        
        u_t_list = np.vstack([p@u.iloc[t+1] +  f*dt for p in P])  # Halmitonian * dt (7.33)
        u_t_list_min = u_t_list.min(axis=0)                                            # min Halmitonian
        a = u_t_list.argmin(axis=0)                                          # corresponding action
        u.iloc[t] = u_t_list_min                                             # record value funtion at t
        a_flow.iloc[t] = a                                                   # record action at t
    return a_flow, u

def solve_fixpoint(0, param_q, param_f, T=10, n=1000, iterations=1):
    tline = np.linspace(0, T, n+1)
    dt = T/n
    _flow = pd.DataFrame(np.random.dirichlet(np.ones(len(0)), size=len(tline)),
                          index=tline, columns=0.index)

    a_flow_list = []
    _flow_list = []
    for i in range(iterations):
        a_flow, _ = solve_hjb(_flow, param_q, param_f)
        _flow = solve_flow(0, a_flow, param_q, param_f)


        _flow_list.append(_flow)
        a_flow_list.append(a_flow)
    
    return (_flow_list, a_flow_list)
